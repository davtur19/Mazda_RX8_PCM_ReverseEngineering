/*
 * Single ADC Conversion with Synchronous Read
 * 
 * Purpose: Execute one complete ADC conversion cycle with hardware polling
 * 
 * Parameters:
 *   R4 = timeout value (0xFF = full wait, reduced = faster timeout)
 *
 * Hardware Sequence:
 *   1. Clear PGDR conversion ready flag
 *   2. Wait 40 CPU cycles (ATU synchronization delay)
 *   3. Set SCI0_SSR to initiate A/D conversion request
 *   4. Load channel select to SCI0_TDR
 *   5. Poll SCI0_SSR RDRF (Receive Data Ready Flag) with timeout
 *   6. Read SCI0_RDR (ADC result, 10-bit unsigned)
 *   7. Set PGDR conversion complete flag
 *   8. Return ADC result in R0
 *
 * Timing: ~100 microseconds per conversion
 * Conversion Speed: 10-bit SAR ADC, ~100 kHz sampling
 */

uint16_t adc_single_convert(uint8_t timeout) {
    // R4 = timeout value
    
    // Delay 40 CPU cycles for hardware settling
    nop_delay_40cycles();
    
    // Clear PGDR bit 1 (conversion ready flag)
    PGDR &= ~0x0002;  // bitfield_set_or_clear(PGDR, 0x0001, 0)
    
    // Wait another 40 cycles
    nop_delay_40cycles();
    
    // Set SCI0 Status Register for A/D conversion
    SCI0_SSR = (SCI0_SSR & 0x87) | 0x80;  // Set TE (Transmit Enable)
    
    // Write channel selection to transmit data register
    uint8_t channel = R2;  // R2 contains channel ID
    SCI0_TDR = channel;
    
    // Set SSR to receive state: CKE=11, RIE=1, TE=1, RE=1
    SCI0_SSR = (SCI0_SSR & 0x7F) | 0x78;
    
    // Poll RDRF (bit 6 = 0x40) with timeout
    uint8_t status_byte;
    uint32_t poll_count = 0;
    do {
        status_byte = SCI0_SSR;
        poll_count++;
    } while ((status_byte & 0x40) == 0 && poll_count < timeout);
    
    // Read ADC result from receive data register (10-bit value)
    uint16_t adc_result = SCI0_RDR;
    
    // Set conversion complete in PGDR
    PGDR &= 0xBF;
    PGDR |= 0xB8;
    
    // Wait 40 more cycles for ADC to settle
    nop_delay_40cycles();
    
    // Set PGDR bit 1 again (conversion done flag)
    PGDR |= 0x0002;  // bitfield_set_or_clear(PGDR, 0x0001, 1)
    
    return adc_result;  // Return in R0
}

    uint8_t channel_index = 0;

    uint32_t sensor_base = 0xFFFFA0D8;

        // Calculate offset: index * 40 bytes

            sensor_channel_update(channel_index);

    }

/*
 * Sensor Value Read with Filtering
 * 
 * Purpose: Read raw ADC sensor value and apply first-stage filtering
 *
 * Parameters:
 *   R4 = sensor channel ID (0-3)
 *
 * Processing:
 *   1. Check task priority for channel access
 *   2. Load sensor control flags from RAM
 *   3. If sensor disabled (bit 5 = 1), skip filtering
 *   4. Get ADC min/max thresholds and current reading
 *   5. Compare ADC value against valid range
 *   6. If out of range, flag error; else update filtered value
 *
 * Error Conditions:
 *   - ADC reading below minimum threshold -> Flag error state
 *   - ADC reading above maximum threshold -> Flag error state  
 *   - Hardware timeout in conversion -> Flag error state
 *
 * RAM Addresses:
 *   0xFFFFA0D8: Sensor channel 0 config
 *   0xFFFFA118: Sensor channel 1 config
 *   0xFFFFA158: Sensor channel 2 config
 *   0xFFFFA198: Sensor channel 3 config
 *   Each channel: 40-byte structure with ADC min/max at offsets +10/+14
 */

void sensor_read_and_filter(uint8_t channel_id) {
    // R4 = channel_id
    
    // Check priority
    task_check_priority_level(0x10);
    
    // Calculate sensor config address: base + (channel * 40)
    uint32_t sensor_base = 0xFFFFA0D8;
    uint32_t channel_offset = (channel_id * 40);
    uint32_t sensor_config = sensor_base + channel_offset;
    
    // Check if sensor is disabled (byte offset +5 bit 7)
    uint8_t control_flags = *(uint8_t*)(sensor_config + 5);
    if (control_flags != 0) {
        // Sensor disabled, skip filtering
        return;
    }
    
    // Load sensor ADC limits from config
    uint32_t adc_range_entry = *(uint32_t*)(sensor_config + 0xC);
    uint16_t adc_min = *(uint16_t*)adc_range_entry;  // +3 for alignment
    
    uint32_t adc_current_entry = *(uint32_t*)(sensor_config + 0x4);
    uint16_t adc_current = *(uint16_t*)adc_current_entry;
    
    // Check if ADC reading is valid
    int16_t delta = adc_current - adc_min;  // Subtraction with sign extend
    
    if (delta < 0) {
        // Reading below minimum - invalid
        // Do not update filtered value
        goto skip_update;
    }
    
    // Check upper threshold
    uint32_t adc_max_entry = *(uint32_t*)(sensor_config + 0x10);
    uint16_t adc_max = *(uint16_t*)(adc_max_entry);
    uint16_t max_flags = *(uint16_t*)(sensor_config + 0x14);
    
    if ((max_flags & adc_current) != 0) {
        // Check flag bits
        goto skip_update;
    }
    
    // ADC value is valid - update filtered result
    float filtered_value = 0.0;  // R5 = 0.0
    write_to_memory_with_fpu_sync(
        *(uint32_t*)sensor_config,  // Destination (RAMoffset +0)
        filtered_value
    );
    
    // Update both channel state bytes
    *(uint8_t*)(sensor_config + 4) = 0;  // Clear processing flag
    *(uint8_t*)(sensor_config + 5) = 0;  // Clear error flag
    
skip_update:
    task_conditional_dispatch(priority_context);
}

/*
 * ADC Channel Chain Configuration for Multi-Channel Sampling
 * 
 * Purpose: Configure ATU2 timer channels for synchronized ADC sampling
 *          of all sensor channels in a predefined sequence
 *
 * Channels Configured (in order):
 *   ATU2A: Primary sensor (TPS or MAF)
 *   ATU2B: Secondary sensor (CLT or O2)
 *   ATU2 Channels 2-8: Additional sensor/output synchronization
 *
 * Hardware Registers (ATU2 = Address Time Unit 2):
 *   ATUII_TIOR2A (0xFFFFA2C8): I/O Control Register, Channel A
 *   ATUII_TIOR2B (0xFFFFA2CA): I/O Control Register, Channel B  
 *   ATUII_TIER2A (0xFFFFA2CC): Interrupt Enable Register, Channel A
 *   ATUII_TIER8 (0xFFFFA2D4): Interrupt Enable Register, Channel 8
 *   ATUII_TIOR2C (0xFFFFA2D0): Additional trigger config
 *
 * Configuration Sequence:
 *   1. Clear prior interrupt requests
 *   2. Set TIOR2A output compare mode = compare output (bits 5-3 = 001)
 *   3. Set TIOR2B output compare mode = compare output (bits 5-3 = 001)
 *   4. Disable timer interrupts for channels (clear TIER bits)
 *   5. Clear any pending interrupt flags
 *   6. Reset channel mode bits (TIOR bits 7-6 = 00)
 */

void sensor_adc_convert_chain(void) {
    // Initialize FPU
    fpu_load_zero(FPU_TEMP_STACK);
    
    uint16_t atuii_tior2a = ATUII_TIOR2A_ADDR;  // 0xFFFFA2C8
    uint16_t atuii_tier2a = ATUII_TIER2A_ADDR;  // 0xFFFFA2CC
    uint16_t atuii_tier8 = ATUII_TIER8_ADDR;    // 0xFFFFA2D4
    uint16_t atuii_tior2b = ATUII_TIOR2B_ADDR;  // 0xFFFFA2D0
    
    // Configure Channel 2A output mode
    uint8_t tior2a_val = *(uint8_t*)atuii_tior2a;
    tior2a_val = (tior2a_val & 0xF8) | 0x01;    // Bits 2-0 = 001
    *(uint8_t*)atuii_tior2a = tior2a_val;
    
    // Disable TIER2A interrupts
    uint16_t tier2a_val = *(uint16_t*)atuii_tier2a;
    tier2a_val &= ~0xFFFE;  // Clear all interrupt bits except bit 0
    *(uint16_t*)atuii_tier2a = tier2a_val;
    
    // Disable TIER8 interrupts
    uint16_t tier8_val = *(uint16_t*)atuii_tier8;
    tier8_val &= ~0xFEFE;   // Clear pattern
    *(uint16_t*)atuii_tier8 = tier8_val;
    
    // Configure Channel 2B output mode
    uint8_t tior2b_val = *(uint8_t*)atuii_tior2b;
    tior2b_val = (tior2b_val & 0xF8) | 0x01;
    *(uint8_t*)atuii_tior2b = tior2b_val;
    
    // Second configuration pass with different bit pattern
    tier2a_val = *(uint16_t*)atuii_tier2a;
    tier2a_val &= ~0xFFFA;  // Clear different bits
    *(uint16_t*)atuii_tier2a = tier2a_val;
    
    tier8_val = *(uint16_t*)atuii_tier8;
    tier8_val &= ~0xFDFE;
    *(uint16_t*)atuii_tier8 = tier8_val;
    
    // Configure TIOR2A bits 7-6 = 0001 (output mode)
    tior2a_val = *(uint8_t*)atuii_tior2a;
    tior2a_val = (tior2a_val & 0x8F) | 0x10;
    *(uint8_t*)atuii_tior2a = tior2a_val;
    
    // Final interrupt disable pass
    tier2a_val = *(uint16_t*)atuii_tier2a;
    tier2a_val &= ~0xFFFC;
    *(uint16_t*)atuii_tier2a = tier2a_val;
    
    tier8_val = *(uint16_t*)atuii_tier8;
    tier8_val &= ~0xFDFE;
    *(uint16_t*)atuii_tier8 = tier8_val;
    
    // Final TIOR2A configuration
    tior2a_val = *(uint8_t*)atuii_tior2a;
    tior2a_val = (tior2a_val & 0x8F) | 0x10;
    *(uint8_t*)atuii_tior2a = tior2a_val;
    
    // Disable remaining interrupts
    tier2a_val = *(uint16_t*)atuii_tier2a;
    tier2a_val &= ~0xFFF6;
    *(uint16_t*)atuii_tier2a = tier2a_val;
    
    tier8_val = *(uint16_t*)atuii_tier8;
    tier8_val &= ~0xF7FE;
    *(uint16_t*)atuii_tier8 = tier8_val;
    
    // FPU sync barrier
    fpu_nop_stub();
}

/*
 * Sensor Interpolation Calculation - Channel 0 (TPS)
 * 
 * Purpose: Perform bilinear interpolation on 2D lookup table
 *          Maps ADC raw value to physical sensor output
 *
 * Input: ADC raw value from Channel 0 (Throttle Position)
 *
 * Calibration Tables:
 *   0xFFFF9F1E: Current ADC value register
 *   0x6CF4A: Min X axis value (ADC min)
 *   0xFFFF9F66: Interpolation weight (0.0-1.0)
 *   0x6CF46: Max valid TPS value
 *   0x6CF48: Upper TPS range limit
 *
 * Interpolation Method:
 *   1. Calculate weight = (adc_val - adc_min) / (adc_max - adc_min)
 *   2. Lookup table base at ROM address
 *   3. Linear interpolation: output = base + weight * (next_val - base)
 *   4. Clamp output to valid range (0-100%)
 *
 * Output States:
 *   0: Valid TPS reading
 *   1: TPS above max range threshold
 *   2: TPS below min range threshold
 *
 * RAM Storage: 0xFFFF9F60 (float result), 0xFFFF9F64 (status byte)
 */

void sensor_calc_interp_ch0(void) {
    // Load current ADC value for channel 0
    uint16_t adc_raw = *(uint16_t*)0xFFFF9F1E;
    
    // Load calibration constants
    uint16_t adc_min = *(uint16_t*)0x6CF4A;
    uint16_t table_x_axis = *(uint16_t*)0xFFFF9F66;
    
    // Calculate interpolation weight
    // weight = (adc_raw - adc_min) / (range)
    uint16_t weight = interp_bilinear_weight_calc(adc_raw, table_x_axis, adc_min);
    
    *(uint16_t*)0xFFFF9F66 = weight;  // Store weight for next iteration
    
    // Load float constant for scaling
    float scale_constant = 1.0;  // Loaded from ROM 0x7334
    
    // Perform linear interpolation
    // Calls fpu_multiply_accumulate with address 0x6A14A
    float interp_result = interp_linear_u16_to_float(weight);
    
    // Multiply by calibration constant
    float tps_output = scale_constant * interp_result;
    *(float*)0xFFFF9F60 = tps_output;
    
    // Validate output range
    uint16_t max_threshold_1 = *(uint16_t*)0x6CF46;
    if (weight >= max_threshold_1) {
        // Output exceeds range 1
        *(uint8_t*)0xFFFF9F64 = 1;
        return;
    }
    
    uint16_t max_threshold_2 = *(uint16_t*)0x6CF48;
    if (weight >= max_threshold_2) {
        // Output exceeds range 2
        *(uint8_t*)0xFFFF9F64 = 2;
        return;
    }
    
    // Output is valid
    *(uint8_t*)0xFFFF9F64 = 0;
}

/*
 * Frequency-Based Sensor Calculation - Channel 1 (MAF)
 * 
 * Purpose: Convert frequency input (from MAF hot-wire anemometer) to mass
 *          airflow. MAF sensor outputs variable frequency signal based on
 *          air velocity through sensor element.
 *
 * Input: Frequency counter value (timer captures)
 *
 * Calibration:
 *   0xFFFF9F00: Frequency counter raw value
 *   0x6CF50: Base frequency for idle
 *   0xFFFF9F6C: Current calculation frequency
 *   0x6CF4C: Frequency-to-airflow scale factor (float)
 *   0xFFFF9F68: Calculated airflow (result)
 *
 * Calculation:
 *   1. Read frequency count from hardware timer
 *   2. Calculate weight: (freq - freq_idle) / (freq_max - freq_idle)
 *   3. Perform interpolation with bilinear weight
 *   4. Scale by constant: airflow = weight * scale_factor
 *   5. Convert to fixed-point (integer g/s)
 *   6. Store result in RAM
 *
 * Output Range: 0-255 g/s (grams per second)
 * Idle Frequency: ~100-150 Hz (calibration dependent)
 * Max Frequency: ~4000-5000 Hz (full load)
 *
 * FPU Operations:
 *   - Float conversion of frequency count
 *   - Division and scaling
 *   - Fixed-point result conversion
 */

void sensor_calc_freq_ch1(void) {
    // Load frequency counter value
    uint16_t freq_count = *(uint16_t*)0xFFFF9F00;
    
    // Load calibration constants  
    uint16_t idle_frequency = *(uint16_t*)0x6CF50;
    uint16_t freq_lookup = *(uint16_t*)0xFFFF9F6C;
    
    // Save frequency for later
    uint16_t saved_freq = freq_count;
    
    // Calculate weight using bilinear interpolation
    uint16_t weight = interp_bilinear_weight_calc(freq_count, freq_lookup, idle_frequency);
    
    // Convert weight to float
    uint32_t weight_int = weight & 0xFFFF;
    float weight_float = (float)weight_int;
    
    // Load scaling constant
    float scale_const = 1.0;  // From ROM 0x73B0
    float scale_factor = *(float*)0x6CF4C;  // Frequency->airflow multiplier
    
    // Perform scaling: airflow = (weight_float / 1000.0) * scale_factor
    float scaled_airflow = (weight_float / 1000.0) * scale_factor;
    
    // Store intermediate result
    *(float*)0xFFFF9F68 = scaled_airflow;
    
    // Store frequency weight for next cycle
    *(uint16_t*)0xFFFF9F6C = weight;
}

 *   3. Perform linear interpolation in temperature lookup table
 *   4. Multiply by scale factor to get degrees C
 *   5. Validate against min/max thresholds
 *   6. Clamp to valid range
 *
 * Output States:
 *   0: Valid temperature reading
 *   1: Temperature below minimum (-40C typical)
 *   2: Temperature above maximum (+125C typical)
 *
 * RAM Locations:
 *   0xFFFF9F70: Final temperature (float, degrees C)
 *   0xFFFF9F74: Status flag (0=OK, 1=low, 2=high)
 */

void sensor_calc_interp_ch2(void) {
    // Load ADC raw value
    uint16_t adc_value = *(uint16_t*)0xFFFF9F1C;
    
    // Load calibration axis
    uint16_t temp_axis = *(uint16_t*)0x6CF56;
    uint16_t table_weight = *(uint16_t*)0xFFFF9F76;
    
    // Calculate interpolation weight
    uint16_t weight = interp_bilinear_weight_calc(adc_value, temp_axis, adc_value);
    
    // Perform linear float interpolation
    float temp_scale_const = 1.0;  // From ROM 0x743C
    float interp_temp = interp_linear_u16_to_float(weight);
    
    // Multiply by constant to get Celsius
    float clt_celsius = temp_scale_const * interp_temp;
    *(float*)0xFFFF9F70 = clt_celsius;
    
    // Store weight for tracking
    *(uint16_t*)0xFFFF9F76 = weight;
    
    // Validate temperature range
    uint16_t min_valid_temp = *(uint16_t*)0x6CF52;
    if (weight >= min_valid_temp) {
        *(uint8_t*)0xFFFF9F74 = 1;  // Below minimum
        return;
    }
    
    uint16_t max_valid_temp = *(uint16_t*)0x6CF54;
    if (weight >= max_valid_temp) {
        *(uint8_t*)0xFFFF9F74 = 2;  // Above maximum
        return;
    }
    
    *(uint8_t*)0xFFFF9F74 = 0;  // Valid range
}

11. SENSOR_CHANNEL_UPDATE (Address: 0x914c)
================================================================================

/*
 * Sensor Channel State Update
 * 
 * Purpose: Perform single update of one sensor channel:
 *          - Read raw ADC value
 *          - Apply basic validation
 *          - Optionally apply low-pass filter
 *          - Store in output register
 *
 * Parameters: R4 = channel_id (0-3)
 *
 * Processing:
 *   1. Calculate sensor config address = 0xFFFFA0D8 + (ch_id * 40)
 *   2. Check if interpolation is needed (byte +5 test)
 *   3. Load raw ADC from channel storage
 *   4. Apply outlier detection vs. previous value
 *   5. If valid: apply first-order low-pass filter or direct update
 *   6. Write filtered value to output RAM location
 *
 * Filter Type: Single-pole IIR (exponential moving average)
 *   filtered_new = alpha * adc_raw + (1-alpha) * filtered_old
 *   Where alpha = 0.5 (50% weighting) typical
 *
 * RAM Locations (per channel, offset +40 bytes):
 *   +0x00: Filtered output value (float)
 *   +0x04: Raw ADC value (uint16)
 *   +0x08: Filter coefficient storage
 *   +0x10: Min threshold pointer
 *   +0x14: Output register address
 *   +0x20: Filter weight (0x3E80 = 0.25 float)
 *
 * Outlier Detection Thresholds:
 *   Max delta from previous = 0.5 units (hysteresis)
 */

void sensor_channel_update(uint8_t channel_id) {
    // R4 = channel_id
    
    // Calculate sensor config address  
    uint32_t sensor_base = 0xFFFFA0D8;
    uint32_t config_addr = sensor_base + (channel_id * 40);
    
    // Check if interpolation is disabled (byte +5)
    uint8_t control_byte = *(uint8_t*)(config_addr + 5);
    if (control_byte == 0) {
        // Skip update if interpolation enabled
        goto skip_channel;
    }
    
    // Load current ADC value
    float current_adc_float = *(float*)(config_addr + 0xC);
    float last_adc_value = *(float*)(config_addr + 0xFC);
    
    // Calculate difference for outlier detection
    float adc_delta = current_adc_float - last_adc_value;
    
    // Outlier threshold constant
    float outlier_threshold = 0.5;  // From ROM 0x924C
    
    if (fabs(adc_delta) > outlier_threshold) {
        // Value changed too much - likely noise/fault
        // Apply filter: new = 0.5 * current + 0.5 * last
        float filter_weight = 0.25;  // From ROM 0xC350
        sensor_filter_apply(channel_id, filter_weight);
        
        // Get filtered result
        uint32_t output_addr = *(uint32_t*)(config_addr + 0x10);
        uint16_t filtered_value = *(uint16_t*)(output_addr + 0x14);
        *(uint16_t*)output_addr = filtered_value;
    } else {
        // Value is within tolerance - store directly
        uint32_t output_addr = *(uint32_t*)(config_addr + 0x10);
        *(uint32_t*)output_addr = 0;  // Write zero (valid)
        
        *(uint8_t*)(config_addr + 4) = 0;
        *(uint8_t*)(config_addr + 5) = 0;
    }
    
skip_channel:
    // Return to task dispatcher
}

12. SENSOR_INTERP_CALC (Address: 0x91fe)
================================================================================

/*
 * Advanced Sensor Interpolation with Multi-Point Lookup
 * 
 * Purpose: Perform complete sensor value interpolation using 2D calibration
 *          tables with RPM and load axis support
 *
 * Parameters: R4 = channel_id (0-3)
 *
 * Calculation Flow:
 *   1. Load raw ADC value for channel
 *   2. Load current engine RPM and load (MAP)
 *   3. Find interpolation weights on both axes
 *   4. Look up table cells: (RPM_idx, Load_idx)
 *   5. Perform bilinear interpolation between 4 corner values
 *   6. Apply axis-dependent scaling
 *   7. Validate result against limits
 *   8. Apply optional low-pass filter for noise rejection
 *
 * Calibration Table Access:
 *   ROM Base (0xDAB4): Sensor channel table pointers
 *   Each table: variable-length with axis definitions
 *
 * RAM Workspace:
 *   0xFFFFA0D8: Sensor channel 0 config base
 *   0xFFFFA0FC: Current float value
 *   0xFFFFA0F8: Engine speed normalization
 *   0xFFFFA100: Engine load value (RPM/4)
 *   0xFFFFA0D4: ADC time base
 *
 * Axis Bounds Check:
 *   If (value < axis_min): Use minimum axis point
 *   If (value > axis_max): Use maximum axis point
 *   Else: Linear interpolate between surrounding points
 *
 * Outlier Detection:
 *   Delta > 0.125 units: Apply 0.5x filter (noisy conditions)
 *   Delta <= 0.125: Direct update (stable)
 *
 * FPU Operations:
 *   - 3 division operations for bilinear weighting
 *   - Floating-point subtraction for range checking
 *   - Conditional clamping based on limits
 */

void sensor_interp_calc(uint8_t channel_id) {
    // R4 = channel_id
    
    // Calculate channel config offset
    uint32_t sensor_base = 0xFFFFA0D8;
    uint32_t ch_offset = (channel_id * 40);
    uint32_t sensor_config = sensor_base + ch_offset;
    
    // Load raw ADC value from sensor
    float raw_adc = *(float*)(sensor_config + 0xFC);
    
    // Load engine parameters
    float engine_rpm_normalized = *(float*)0xFFFFA0F8;  // RPM/1000
    float engine_load = *(float*)0xFFFFA100;  // Normalized load
    
    // Basic sanity check on ADC
    float outlier_threshold = 0.125;  // From ROM 0x924C
    float adc_delta = raw_adc - *(float*)(sensor_config);
    
    if (fabs(adc_delta) > outlier_threshold) {
        // Outlier detected, skip interpolation
        return;
    }
    
    // Load sensor calibration table pointer
    uint32_t table_base_addr = 0xDAB4;
    uint32_t table_entry = *(uint32_t*)(table_base_addr + (channel_id * 4));
    
    // Check if table is valid (non-zero entry)
    uint16_t table_header = *(uint16_t*)table_entry;
    if (table_header == 0) {
        // No table available - set error flags
        *(uint8_t*)(sensor_config + 4) = 0;
        *(uint8_t*)(sensor_config + 5) = 0;
        return;
    }
    
    // Perform RPM axis lookup
    float rpm_weight = *(float*)0xFFFFA0D4;  // Interpolation weight for RPM
    uint16_t rpm_factor = *(uint16_t*)(table_entry + 0xC) & 0xFF;
    
    // Calculate scaled RPM weighting
    float rpm_scale = (float)rpm_factor * 0.0039062;  // Divide by 256
    float rpm_interpolated = rpm_weight / rpm_scale;
    
    // Perform load axis lookup using MAP/pressure
    float load_scale_const = 0.5;  // From ROM 0x946C
    float load_weight = engine_rpm_normalized * load_scale_const;
    
    // Perform 2D table interpolation
    // Get corner values from table: [rpm_idx, load_idx]
    uint16_t table_offset = *(uint16_t*)(table_entry + 0x4);
    float val_low_low = *(float*)(table_base_addr + table_offset);
    float val_high_low = *(float*)(table_base_addr + table_offset + 4);
    
    // Calculate bilinear weights
    float x_weight = rpm_weight;
    float y_weight = load_weight;
    
    // Bilinear: result = (1-x)(1-y)*v00 + x(1-y)*v10 + (1-x)y*v01 + xy*v11
    float result_interpolated = val_low_low;
    result_interpolated += (val_high_low - val_low_low) * x_weight;
    
    // Store interpolated value
    *(float*)(sensor_config + 0xFC) = result_interpolated;
    
    // Validate against output limits
    float max_valid = 127.0;  // From ROM 0x925C
    if (result_interpolated > max_valid) {
        // Clamp to maximum
        *(uint8_t*)(sensor_config + 4) = 1;  // Set error flag
    } else {
        *(uint8_t*)(sensor_config + 4) = 0;  // Clear error flag
    }
}

13. SENSOR_FILTER_APPLY (Address: 0x9478)
================================================================================

/*
 * Low-Pass IIR Filter Application
 * 
 * Purpose: Apply first-order exponential smoothing filter to reduce
 *          sensor noise and jitter
 *
 * Parameters:
 *   R4 = channel_id
 *   R5 = filter weight (0x3E80 = 0.25 typical, 0x3F00 = 0.5)
 *
 * Filter Equation:
 *   y[n] = alpha * x[n] + (1 - alpha) * y[n-1]
 *   Where alpha = filter_weight, x[n] = new sample, y[n-1] = previous output
 *
 * Weights:
 *   0x3E00 (0.125): Heavy filtering, ~8 sample average
 *   0x3E80 (0.25):  Moderate filtering, ~4 sample average  
 *   0x3F00 (0.5):   Light filtering, ~2 sample average
 *
 * Processing:
 *   1. Load previous filtered value from RAM
 *   2. Load current raw sensor value
 *   3. Calculate: new_filtered = (1 - weight) * previous + weight * current
 *   4. Store result back to filter state
 *   5. Copy to output register
 *
 * RAM Addresses (per channel):
 *   sensor_base + 0: Current filtered output
 *   sensor_base + 4: Previous value (IIR state)
 *   sensor_base + 8: Filter coefficient storage
 */

void sensor_filter_apply(uint8_t channel_id, float filter_weight) {
    // R4 = channel_id, R5 = filter weight (float)
    
    uint32_t sensor_base = 0xFFFFA0D8;
    uint32_t config_addr = sensor_base + (channel_id * 40);
    
    // Load previous filtered value (IIR state)
    float previous_filtered = *(float*)(config_addr + 0);
    
    // Load current raw value
    float current_raw = *(float*)(config_addr + 0xFC);
    
    // Calculate: new_filtered = (1.0 - weight) * prev + weight * current
    float one_minus_weight = 1.0 - filter_weight;
    float filtered_new = (one_minus_weight * previous_filtered) + 
                         (filter_weight * current_raw);
    
    // Store back to state
    *(float*)(config_addr + 0) = filtered_new;
    
    // Store in output register
    uint32_t output_addr = *(uint32_t*)(config_addr + 0x10);
    *(float*)output_addr = filtered_new;
}

14. CRANK_POSITION_STATE_MACHINE (Address: 0x789e)
================================================================================

/*
 * Crankshaft Position Synchronization and Angle Tracking
 * 
 * Purpose: Track camshaft/crankshaft position through tooth-by-tooth signal
 *          detection and maintain synchronization with 4-stroke cycle
 *
 * Hardware Input:
 *   ATU2 Capture channels detect rising/falling edges on crank sensor signal
 *   CAS (Crank Angle Sensor) produces pulse train with missing teeth pattern
 *   Mazda RX-8: 36-1 (36 teeth, 1 missing) = 10 degrees per tooth
 *
 * State Machine:
 *   State 0: Unsynchronized (searching for gap)
 *   State 1: Synchronized, coasting (tracking without fuel delivery)
 *   State 2: Synchronized, active (normal combustion)
 *   State 3: Synchronized, locked (high confidence)
 *
 * Tooth Detection Validation:
 *   - Monitor tooth-to-tooth timing (period)
 *   - Detect missing tooth (longer period = gap in pattern)
 *   - Validate period ratio: 1.2-2.0x normal tooth period
 *   - Count consecutive valid teeth before declaring sync
 *
 * RAM Workspace:
 *   0xFFFF9F95: Current tooth count (0-35)
 *   0xFFFF9FC3: Gap pattern offset
 *   0xFFFF9FC5: Sync counter (gap detections)
 *   0xFFFF9FC2: State machine state (0-3)
 *   0xFFFF9FA1: Tooth count in current engine cycle
 *   0xFFFF9FBC: Accumulator for timing calculation
 *   0xFFFF9FC1: Current state byte
 *
 * Processing:
 *   1. Load current tooth position counter
 *   2. If counter > 36: Reset to 0 (wrap around)
 *   3. Add tooth period to 0xFFFF9FBC accumulator
 *   4. Check for missing tooth pattern
 *   5. Validate tooth timing window (1.0-2.5x nominal)
 *   6. Increment gap counter if valid
 *   7. Transition state when certain patterns detected
 *   8. Update stroke/cycle phase information
 */

void crank_position_state_machine(void) {
    uint8_t max_teeth = 36;
    
    // Load current tooth count
    uint8_t tooth_count = *(uint8_t*)0xFFFF9F95;
    
    // Reset if overflow
    if (tooth_count > max_teeth) {
        crank_vars_init();  // Reset all crank variables
    }
    
    // Load state byte
    uint8_t state = *(uint8_t*)0xFFFF9FC1;
    uint8_t gap_offset = *(uint8_t*)0xFFFF9FC3;
    
    // Add gap offset to tooth count for validation
    if (gap_offset != 0) {
        uint8_t adjusted_count = tooth_count + gap_offset;
        if (adjusted_count < max_teeth) {
            tooth_count = adjusted_count;
        }
    }
    *(uint8_t*)0xFFFF9F95 = tooth_count;
    
    // Accumulate tooth period timing
    uint8_t sync_counter = *(uint8_t*)0xFFFF9FC5;
    uint32_t tooth_period = *(uint32_t*)0xFFFF9FBC;
    
    // Increment counter up to 255
    if (sync_counter < 0xFF) {
        sync_counter++;
    }
    *(uint8_t*)0xFFFF9FC5 = sync_counter;
    
    // Update accumulator (accumulates period for averaging)
    float period_float = (float)(tooth_period + 1) & 0x00FFFFFF;
    *(float*)0xFFFF9FBC = period_float * 1.0;  // Scale factor
    
    // State machine: Handle based on current state
    switch (state) {
        case 0:  // Unsynchronized state
            // Look for gap pattern in tooth periods
            if (tooth_count == 0x24) {  // 36 teeth reached
                state = 0;  // Restart
            }
            break;
            
        case 1:  // Synchronized, coasting
            // Validate sync is maintained
            uint8_t sync_threshold = 1;
            uint8_t tooth_position = *(uint8_t*)0xFFFF9FA1;
            
            if (tooth_position >= sync_threshold) {
                // Check for valid tooth period from table at DA04
                uint8_t valid_tooth_period = *(uint8_t*)(0xDA04 + tooth_count);
                
                if (valid_tooth_period != 0) {
                    // Tooth pattern valid, increment state counter
                    state = 1;
                    *(uint8_t*)0xFFFF9FA2 = 0;
                }
            }
            break;
            
        case 2:  // Synchronized, active
            // In combustion phase
            // Lookup missing tooth position from table DA4E
            uint32_t table_offset = 0xDA4E;
            uint8_t gap_position = *(uint8_t*)(table_offset + sync_counter);
            
            uint8_t next_tooth = tooth_count + gap_position;
            
            // Check if tooth position matches expected pattern
            if (next_tooth == 0x1C || next_tooth == 0x0A) {
                // Found missing tooth at expected location
                state = 3;  // Transition to locked state
                *(uint8_t*)0xFFFF9FA2 = 0;
                break;
            }
            
            // Validate sync hasn't been lost
            if (sync_counter >= 0xFF) {
                *(uint8_t*)0xFFFF9FA2 = 1;  // Set error flag
            }
            break;
            
        case 3:  // Synchronized, locked
            // Perform comprehensive tooth validation
            uint32_t timing_threshold = 0xA2C28;
            uint32_t total_timing = *(uint32_t*)0xFFFF9FB8;
            
            if (total_timing >= timing_threshold) {
                // Good tooth spacing consistency
                *(uint8_t*)0xFFFF9FE1 = 1;  // Set lock flag
            }
            
            // Check for additional valid tooth
            if (sync_counter >= 3) {
                // Perform final validation
                uint32_t validation_addr = 0xDA04;
                uint8_t validate_result = crank_tooth_validate(
                    (float*)0x331);  // Validation constant
                
                if (validate_result != 0) {
                    // Tooth validation passed
                    tooth_count = 0x21;  // Reset tooth position
                    *(uint32_t*)0xFFFF9FAC = *(uint32_t*)0xFFFF9FB8;
                }
            }
            break;
    }
}

================================================================================
15. CRANK_PERIOD_CALC (Address: 0x7ff0)
================================================================================

/*
 * Crankshaft Rotation Period and RPM Calculation
 * 
 * Purpose: Calculate engine RPM from tooth-to-tooth timing measurements
 *          Accumulate tooth periods and compute averaged engine speed
 *
 * Parameters:
 *   R4 = tooth_count (0-35 for 36-1 pattern)
 *
 * Calculation Method:
 *   1. Convert tooth count to float
 *   2. Load base timing period from 0xFFFF9FA8
 *   3. Calculate: period_with_offset = base + offset_factor * tooth_count
 *   4. Accumulate in running buffer (circular, 6 entries)
 *   5. Every 6 accumulations: Calculate RPM
 *   6. RPM = 60,000,000 / (accumulated_period_microseconds * 10 deg/tooth)
 *
 * Timing:
 *   Tooth period typically 5-50 microseconds per 10 degrees rotation
 *   At 1000 RPM: Period per tooth ~0.6 ms, all 36 teeth ~21.6 ms
 *   At 8000 RPM: Period per tooth ~75 us, all 36 teeth ~2.7 ms
 *
 * ROM Calibration (0x80DC-0x80F8):
 *   0x80DC: Period calculation constant
 *   0x80E4: Offset multiplier
 *   0x80E8: Correction factor
 *   0x80F8: Final RPM scale divisor
 *
 * RAM Workspace:
 *   0xFFFFA9D0: Circular buffer (6 x 32-bit period values)
 *   0xFFFF9FE8: Buffer fill counter (0-6)
 *   0xFFFF9FE9: Accumulator counter
 *   0xFFFF9F88: Current engine speed (32-bit, units TBD)
 *   0xFFFF9F80: Averaged RPM result
 *
 * Algorithm:
 *   For each tooth:
 *     1. Add to accumulator with offset
 *     2. Store in circular buffer
 *     3. When buffer full:
 *         - Sum all 6 period values
 *         - Divide by 6 for average
 *         - Calculate RPM = 1000000 / avg_period
 *         - Clear buffer counter, restart
 */

uint32_t crank_period_calc(uint8_t tooth_count) {
    // R4 = tooth_count (float conversion)
    
    float tooth_count_f = (float)tooth_count;
    
    // Load base timing period
    uint32_t base_period = *(uint32_t*)0xFFFF9FA8;
    uint32_t total_timing = *(uint32_t*)0xFFFF9FB8;
    
    // Load timing constants from ROM
    float const_period = 1.0;  // From 0x80DC
    float const_offset = 1.0;  // From 0x80E4
    
    // Calculate period with tooth count offset
    // period_adjusted = base_period + offset_factor * tooth_count
    float period_with_adjustment = const_period + 
                                   (const_offset * tooth_count_f);
    
    // Load and store normalized period
    *(uint32_t*)0xFFFF9F84 = base_period;
    *(uint32_t*)0xFFFF9F9C = total_timing;
    
    // Correction for zero crossing or wraparound
    float correction_const = 2000.0;  // From 0x80E8
    if (period_with_adjustment < 0.0) {
        period_with_adjustment = period_with_adjustment + correction_const;
    }
    
    // Store adjusted period
    *(float*)0xFFFF9F98 = period_with_adjustment;
    
    // Manage circular buffer filling
    uint8_t buffer_counter = *(uint8_t*)0xFFFF9FE9;
    buffer_counter++;
    if (buffer_counter >= 6) {
        buffer_counter = 0;
    }
    *(uint8_t*)0xFFFF9FE9 = buffer_counter;
    
    // Get counter and check if we need to compute average
    uint8_t accum_counter = *(uint8_t*)0xFFFF9FE8;
    if (accum_counter >= 0xFF) {
        accum_counter = 0;
    } else {
        accum_counter++;
    }
    *(uint8_t*)0xFFFF9FE8 = accum_counter;
    
    // Store period in circular buffer
    uint32_t buffer_offset = buffer_counter << 2;  // *4 for 32-bit entries
    uint32_t buffer_addr = 0xFFFF9FD0;
    *(uint32_t*)(buffer_addr + buffer_offset) = (uint32_t)period_with_adjustment;
    
    // When buffer counter rolls over, calculate RPM
    if (buffer_counter == 0 && accum_counter < 6) {
        // Sum all 6 period values in buffer
        uint32_t total_period = 0;
        for (int i = 0; i < 6; i++) {
            total_period += *(uint32_t*)(buffer_addr + (i << 2));
        }
        
        // Average period = sum / 6
        uint32_t avg_period = total_period;
        
        // Store intermediate result
        *(uint32_t*)0xFFFF9F88 = total_period;
        
        // Calculate RPM = constant / avg_period
        // RPM scale divisor at 0x80F8
        float rpm_divisor = 1000000.0;  // From 0x80F8
        float rpm_result = rpm_divisor / (float)avg_period;
        
        *(float*)0xFFFF9F80 = rpm_result;
    }
    
    return base_period;
}

================================================================================
16. CRANK_TOOTH_VALIDATE (Address: 0x7ad6)
================================================================================

/*
 * Crankshaft Tooth Signal Validation
 * 
 * Purpose: Validate incoming tooth signal against expected period and ratio
 *          Reject noise/glitches and confirm proper crank pattern
 *
 * Input:
 *   R4 = validation threshold (0.0-1.0 confidence level)
 *   FR4 = test period (float, in microseconds)
 *
 * Validation Checks:
 *   1. Check tooth period is positive (timer didn't wrap)
 *   2. Multiply period by test threshold (0.5-2.0x for gap detection)
 *   3. Compare against ROM limit constants
 *   4. If period too short or too long, return error
 *   5. If in valid window, return success (R0 = 0)
 *   6. If accumulated period exceeds accumulator max, flag error
 *
 * Thresholds (ROM 0x7B70-0x7B78):
 *   0x7B70: Minimum tooth period (multiplied result)
 *   0x7B74: Minimum gap period (longer than normal tooth)
 *   0x7B78: Gap timing correction offset
 *
 * Typical Values:
 *   Normal tooth period: 10-50 microseconds (varies with RPM)
 *   Gap period: 1.2-2.5x normal (missing tooth pattern)

 * Return Value: R0
 *   0 = Valid tooth (period acceptable)
 *   1 = Invalid tooth (period out of range)
 */

    // R4 = test_threshold (in float FR4)
    
    uint32_t period_accumulator = *(uint32_t*)0xFFFF9FB4;
    
    // Convert to float
    float period_f = (float)period_accumulator;
    
    // Check for negative period (timer wrap-around)
    if (period_accumulator < 0) {
        // Add wraparound correction (2^32)
        period_f = period_f + 4294967296.0;
    }
    
    // Multiply by test threshold to get weighted period
    float weighted_period = period_f * test_threshold;  // FR5 = FR4 * FR3
    
    // Load ROM limits
    float min_limit = 1.0;  // From ROM 0x7B70
    float max_limit = 2.0;  // From ROM 0x7B74
    
    // Validate weighted period is in range
    // min_limit <= weighted_period <= max_limit
    if (weighted_period < max_limit) {
        // Below minimum threshold - likely not a real tooth
        return 1;  // Invalid
    }
    
    // Check upper bound
    if (weighted_period < min_limit) {
        return 1;  // Still too low
    }
    
    // Round to integer for final check
    uint32_t period_int = (uint32_t)weighted_period;
    
    // Check against accumulator limit
    uint32_t accum_max = *(uint32_t*)0xFFFF9FB0;
    if (period_int >= accum_max) {
        // Accumulation too high, flag error
        *(float*)0xFFFF9FBC = *(float*)0xFFFF9FBC + 1.0;  // Increment error count
        return 1;
    }
    
    return 0;  // Valid tooth period
}

 *   2. Check if desired message ID already queued at that mailbox
 *   3. If queued, return without re-enqueueing (duplicate prevention)
 *   4. Increment mailbox pointer (wrap at 4)

 *   6. Write message data to mailbox + 4
 *   7. Clear command register to initiate transmission
 *   8. Increment pointer for next call

 *   - All mailbox offsets calculated as (index << 3) for 8-byte stride
 *   - Mailbox pointer wraps: 0->1->2->3->0 (4 mailboxes)
 *   - Message ID comparison prevents duplicate messages in flight
 *   - Data field (R5) typically contains encoded message payload
 *   - Called at 10 Hz from can_periodic_task_handler()
 *
 */

void hcan0_mailbox_rotate_enqueue(uint16_t msg_id, uint32_t msg_data)
{
    volatile uint32_t *mailbox_base = (volatile uint32_t *)0xFFFF8004;
    volatile uint8_t *mailbox_ptr = (volatile uint8_t *)0xFFFF802D;
    volatile uint8_t *mailbox_cmd = (volatile uint8_t *)0xFFFF802C;

    uint8_t current_index;
    uint32_t current_msg_id;
    uint32_t mailbox_offset;
    uint32_t *mailbox_addr;

    // === Read Current Mailbox Index ===
    current_index = *mailbox_ptr;
    current_index &= 0xFF;

    // === Check for Duplicate Message ===
    // Calculate mailbox address: base + (index << 3)
    mailbox_offset = (current_index << 2) + (current_index << 3);  // index * 8
    mailbox_addr = (uint32_t *)((uint32_t)mailbox_base + mailbox_offset);
    current_msg_id = *mailbox_addr;

    if (current_msg_id == msg_id) {
        // Message already queued, return without requeue
        return;
    }

    // === Advance Mailbox Pointer ===
    current_index = *mailbox_ptr;
    current_index &= 0xFF;
    if (current_index >= 4) {
        // Wrap at 4 mailboxes
        *mailbox_ptr = 0;
    } else {
        // Increment and store
        current_index++;
        *mailbox_ptr = current_index;
    }

    // === Calculate Next Mailbox Offset ===
    current_index = *mailbox_ptr;
    current_index &= 0xFF;
    mailbox_offset = (current_index << 2) + (current_index << 3);  // index * 8
    mailbox_addr = (uint32_t *)((uint32_t)mailbox_base + mailbox_offset);

    // === Write Message ID to Mailbox ===
    *mailbox_addr = msg_id;

    // === Write Message Data (at offset +4 in mailbox) ===
    uint32_t *data_addr = (uint32_t *)((uint32_t)mailbox_addr + 4);
    *data_addr = msg_data;

    // === Clear Mailbox Command Register ===
    // Writing 0 clears pending and initiates transmission
    *mailbox_cmd = 0;
}

6. CAN TX MESSAGE ABORT AND RETRY (Address 0x9CC6)
================================================================================

/*
 * can_tx_abort_and_retry(void *mailbox_config)
 *
 * PURPOSE:
 *   Abort current CAN transmission in a mailbox and retry sending.
 *   Checks if mailbox has pending TX, writes abort request to ABACK
 *   register, then writes TX Cancel Request to TXCR register.
 *   Used for timeout recovery and message priority handling.
 *
 * INPUTS:
 *   R4 = pointer to mailbox configuration structure
 *        Contains: [status_byte, padding, padding, padding,
 *                   config_byte, ...]
 *
 * OUTPUTS:
 *   Side effects: writes to HCAN0 ABACK_W and TXCR_W registers
 *   Returns through task_conditional_dispatch() tail call
 *
 * MAILBOX CONFIG STRUCTURE:
 *   Offset +5: Mailbox selector (which HCAN0/HCAN1 mailbox)
 *
 * KEY REGISTERS:
 *   HCAN0_TXPR_W  : TX Pending Register (0xFFFFxxxx)
 *   HCAN0_ABACK_W : Abort Acknowledge Register
 *   HCAN0_TXCR_W  : TX Cancel Request Register
 *
 * ALGORITHM:
 *   1. Call task_check_priority_level(0x90)
 *   2. Get mailbox selector from config[5]
 *   3. Read HCAN0_TXPR_W to check if TX is pending (bit set for mailbox)
 *   4. Get mailbox configuration via can_get_mailbox_config()
 *   5. AND with TX pending mask; if result != 0, TX is pending
 *   6. If pending: write ABACK register with mailbox config
 *   7. If pending: write TXCR register with mailbox config
 *   8. Jump to task_conditional_dispatch() for next task
 *
 * NOTES:
 *   - Priority level check (0x90) ensures system is in valid state
 *   - ABACK write must precede TXCR write for proper abort sequence
 *   - Task dispatcher called via jmp @r3 (tail call, no return)
 *   - Used internally by CAN_TX_QUEUE_RETRY_ALL() to recover stuck messages
 *   - Big-endian register layout requires 16-bit aligned access
 *
 */

void can_tx_abort_and_retry(void *mailbox_config)
{
    volatile uint8_t *config_ptr = (volatile uint8_t *)mailbox_config;
    volatile uint16_t *hcan_txpr = (volatile uint16_t *)HCAN0_TXPR_W;
    volatile uint16_t *hcan_aback = (volatile uint16_t *)HCAN0_ABACK_W;
    volatile uint16_t *hcan_txcr = (volatile uint16_t *)HCAN0_TXCR_W;

    uint16_t txpr_status, mailbox_config_val;
    uint8_t mailbox_id;
    uint16_t priority_level;

    // === Check Task Priority ===
    // Ensure system is in valid scheduler state (priority = 0x90)
    priority_level = task_check_priority_level(0x90);

    // === Get Mailbox Selector ===
    mailbox_id = config_ptr[5];

    // === Get Current TX Pending Status ===
    // Read HCAN0_TXPR to check if this mailbox has TX pending
    txpr_status = *hcan_txpr;

    // === Get Mailbox Configuration ===
    mailbox_config_val = can_get_mailbox_config(mailbox_id);
    mailbox_config_val &= 0xFFFF;

    // === Check if TX is Pending ===
    // Mask TXPR with mailbox config; non-zero means TX pending
    if ((txpr_status & mailbox_config_val) != 0) {
        // === Abort Current Transmission ===

        // Get mailbox offset for ABACK register
        volatile uint16_t *aback_addr =
            can_get_mailbox_offset_high(mailbox_id, HCAN0_ABACK_W);

        // Get mailbox configuration for abort request
        mailbox_config_val = can_get_mailbox_config(mailbox_id);

        // Write abort request to ABACK register
        *aback_addr = mailbox_config_val;

        // === Send TX Cancel Request ===

        // Get mailbox offset for TXCR register
        volatile uint16_t *txcr_addr =
            can_get_mailbox_offset_high(mailbox_id, HCAN0_TXCR_W);

        // Get mailbox configuration for cancel request
        mailbox_config_val = can_get_mailbox_config(mailbox_id);

        // Write cancel request to TXCR register
        *txcr_addr = mailbox_config_val;
    }

    // === Task Dispatch (Tail Call) ===
    // Jump to task_conditional_dispatch() without returning
    task_conditional_dispatch();
}

7. CAN TX MESSAGE PACKING (Address 0xCEF4)
================================================================================

/*
 * can_pack_tx_msg_copy(
 *     uint8_t mailbox_id,
 *     uint8_t msg_id_byte,
 *     uint8_t dlc_mode,
 *     void *payload_ptr
 * )
 *
 * PURPOSE:
 *   Pack a CAN message into HCAN0 mailbox format. Handles mailbox ID
 *   encoding, DLC (Data Length Code) configuration, and copies payload
 *   data. Used for all TX message preparations before transmission.
 *
 * INPUTS:
 *   R4 = mailbox_id (0-15 for HCAN0, selects target mailbox)
 *   R5 = message ID byte (CAN frame identifier)
 *   R6 = DLC/mode (0-8 = data bytes, bits specify frame type)
 *   R7 = payload_ptr (pointer to 8-byte message data)
 *
 * OUTPUTS:
 *   Side effects: writes to HCAN0 mailbox at calculated offset
 *   Payload data copied from *payload_ptr to mailbox data area
 *
 * MAILBOX REGISTERS:
 *   HCAN0_MD0[1]_B : Mailbox data registers (MD0 = low, MD1 = high)
 *
 * DATA LAYOUT (16-byte mailbox):
 *   Bytes 0-1:   Message ID (11-bit standard or 29-bit extended)
 *   Byte 2:      DLC (Data Length Code, 0-8 bytes)
 *   Byte 3:      Control flags
 *   Bytes 4-11:  Payload data (up to 8 bytes)
 *   Bytes 12-15: Status/control
 *
 * ALGORITHM:
 *   1. Save input parameters to stack frame
 *   2. Call can_get_mailbox_offset_high() to get mailbox register address
 *   3. Clear FPU context before SFR operations
 *   4. Calculate exact mailbox offset: base + (mailbox_id & 0xF) * 8
 *   5. Set up bounded_byte_copy() to copy payload
 *   6. Copy DLC and message ID to mailbox
 *   7. Copy payload data (up to 8 bytes) to mailbox data area
 *   8. Update mailbox status register
 *   9. Clear FPU context after operations
 *
 * BOUNDED_BYTE_COPY PARAMETERS:
 *   R4 = mailbox offset address (destination)
 *   R5 = DLC/mode byte
 *   R6 = payload pointer
 *   R7 = ???
 *
 * NOTES:
 *   - All SFR accesses wrapped with fpu_load_zero() and fpu_nop_stub()
 *   - Mailbox offset calculation: (mailbox_id & 0xF) << 3
 *   - DLC field specifies number of payload bytes (0-8)
 *   - Message ID stored in big-endian format in HCAN0_MD0
 *   - Multiple mailboxes can be configured independently
 *   - Used by hcan0_mailbox_rotate_enqueue() and direct TX functions
 *
 */

void can_pack_tx_msg_copy(
    uint8_t mailbox_id,
    uint8_t msg_id_byte,
    uint8_t dlc_mode,
    void *payload_ptr
)
{
    volatile uint8_t *hcan0_md = (volatile uint8_t *)HCAN0_MD0_B;

    uint32_t mailbox_offset;
    volatile uint8_t *mailbox_addr;
    uint8_t mailbox_index;

    // === Save Parameters to Stack ===
    uint8_t stack_mailbox = mailbox_id;
    uint8_t stack_msg_id = msg_id_byte;
    uint8_t stack_dlc = dlc_mode;
    void *stack_payload = payload_ptr;

    // === Get Mailbox High Offset ===
    // Call can_get_mailbox_offset_high(mailbox_id)
    // Returns address of mailbox control register
    volatile uint8_t *mailbox_high = can_get_mailbox_offset_high(mailbox_id);

    // Clear FPU context before SFR operations
    fpu_load_zero(stack_addr, 0xE0);

    // === Calculate Mailbox Offset ===
    // Offset = mailbox_id * 8 (each mailbox is 8 bytes)
    mailbox_index = stack_mailbox & 0x0F;  // Mask to 4-bit index
    mailbox_offset = (mailbox_index << 3);  // Multiply by 8
    mailbox_addr = (volatile uint8_t *)((uint32_t)mailbox_high + mailbox_offset);

    // === Setup FPU Context ===
    fpu_load_zero(stack_addr + 4, 0xE0);

    // === Setup Bounded Byte Copy ===
    volatile uint8_t *target_addr = mailbox_addr;
    uint8_t dlc_value = stack_dlc;
    void *src_payload = stack_payload;
    uint8_t mode_byte = stack_msg_id;

    // Call bounded_byte_copy to copy payload
    // bounded_byte_copy(target, dlc, src, mode)
    bounded_byte_copy(target_addr, dlc_value, src_payload, mode_byte);

    // Clear FPU context after operations
    fpu_nop_stub();
}

8. CAN MAILBOX RX MESSAGE EXTRACTION (Address 0xCFF6)
================================================================================

/*
 * can_mailbox_extract_msg_data(uint8_t mailbox_id)
 *
 * PURPOSE:
 *   Extract CAN message ID and data from received message in mailbox.
 *   Handles both standard 11-bit and extended 29-bit frame IDs.
 *   Decodes compact mailbox format back to standard CAN message format.
 *
 * INPUTS:
 *   R4 = mailbox_id (0-15, selects target HCAN0 mailbox)
 *
 * OUTPUTS:
 *   R0 = decoded CAN message ID (32-bit, lower 29 bits used for extended)
 *        Standard frame: bits [10:0] contain ID
 *        Extended frame: bits [28:0] contain ID
 *
 * MAILBOX REGISTERS:
 *   HCAN0_MC0[1]_B : Mailbox control/data registers
 *
 * MESSAGE FORMAT DECODING:
 *   Standard 11-bit frame:
 *   - Control byte bit 3 (IDE) = 0
 *   - ID encoded in compact format, shifted right 5 bits
 *   - Max ID value = 0x7FF (2048 addresses)
 *
 *   Extended 29-bit frame:
 *   - Control byte bit 3 (IDE) = 1
 *   - ID encoded in 4 bytes: byte4, byte5, byte6, byte7
 *   - Multi-byte bit manipulation to extract 29-bit ID
 *   - Masks applied at multiple stages to isolate bits
 *
 * ALGORITHM:
 *   1. Save mailbox_id to stack
 *   2. Call can_get_mailbox_offset_high() to get mailbox base address
 *   3. Calculate offset: (mailbox_id & 0xF) * 8
 *   4. Read control byte at offset+4 to check IDE bit (bit 3)
 *   5. IF standard frame (IDE=0):
 *        - Shift data right 5 bits and mask with 0x7FF
 *        - Extract bits from byte5, shift left 2x
 *        - AND with mask 0x7F8
 *   6. IF extended frame (IDE=1):
 *        - Read bytes 4-7 and reconstruct 29-bit ID
 *        - Byte4: shift left 8x, AND with 0x1C0000
 *        - Byte5: shift left 2x + 2x + 1x, AND with 0x1C0000
 *        - Byte6: shift left 16x + 2x + 2x, AND with 0x1FE00000
 *        - Byte7: shift left 8x, AND with 0xFF00
 *        - OR all components together
 *   7. Return reconstructed message ID in R0
 *
 * REGISTER USAGE:
 *   R4 = mailbox address
 *   R5 = extracted ID (result)
 *   R6 = temporary byte values
 *   R2 = mask values
 *   R3 = shift/mask constants
 *   R0 = final result (returned)
 *
 * NOTES:
 *   - All shifts are logical (not sign-extended)
 *   - Masking applied to isolate specific bit ranges
 *   - Both 11-bit and 29-bit frame formats supported
 *   - Called after CAN RX interrupt detects message ready
 *   - Result can be used as key to message handler table
 *   - Compact encoding saves RAM but requires bit reconstruction
 *
 */

uint32_t can_mailbox_extract_msg_data(uint8_t mailbox_id)
{
    volatile uint8_t *hcan0_mc = (volatile uint8_t *)HCAN0_MC0_B;

    uint32_t mailbox_offset;
    volatile uint8_t *mailbox_addr;
    uint8_t control_byte;
    uint8_t id_byte4, id_byte5, id_byte6, id_byte7;
    uint32_t extracted_id;
    uint8_t is_extended;

    // === Save Mailbox ID ===
    uint8_t stack_mailbox = mailbox_id;

    // === Get Mailbox Offset ===
    volatile uint8_t *mailbox_high = can_get_mailbox_offset_high(mailbox_id);

    mailbox_id = stack_mailbox & 0x0F;  // Mask to 4-bit index
    mailbox_offset = (mailbox_id << 3);  // Multiply by 8
    mailbox_addr = (volatile uint8_t *)((uint32_t)mailbox_high + mailbox_offset);

    // === Read Control Byte ===
    // Byte at offset+4 contains IDE (Identifier Extension) bit
    control_byte = mailbox_addr[4];
    control_byte &= 0xFF;

    // === Check IDE Bit (bit 3) ===
    is_extended = (control_byte & 0x08);  // Non-zero if extended frame

    if (!is_extended) {
        // === STANDARD 11-BIT FRAME ===
        uint8_t id_data = mailbox_addr[4];
        id_data = (id_data >> 5) & 0x07;  // Extract bits [7:5]

        uint8_t id_byte5 = mailbox_addr[5];
        id_data = (id_byte5 << 2) & 0x7F8;  // Shift left 2, mask [10:3]

        extracted_id = id_data & 0x7FF;  // Mask to 11 bits

    } else {
        // === EXTENDED 29-BIT FRAME ===
        uint32_t result = 0;

        // Read raw bytes
        id_byte4 = mailbox_addr[4];
        id_byte5 = mailbox_addr[5];
        id_byte6 = mailbox_addr[6];
        id_byte7 = mailbox_addr[7];

        // Byte 4 extraction (bits [28:21])
        uint32_t byte4_val = id_byte4;
        byte4_val = (byte4_val << 8) & 0x1C0000;

        // Byte 5 extraction (bits [20:13])
        uint32_t byte5_val = id_byte5;
        byte5_val = (byte5_val << 2) & 0x1C000;
        byte5_val = (byte5_val << 2) & 0x1C000;

        // Byte 6 extraction (bits [12:5])
        uint32_t byte6_val = id_byte6;
        byte6_val = (byte6_val << 16) & 0x1FE00000;
        byte6_val = (byte6_val << 2) & 0x1FE00000;

        // Byte 7 extraction (bits [4:0])
        uint32_t byte7_val = id_byte7;
        byte7_val = (byte7_val << 8) & 0xFF00;

        // OR all components
        result = byte4_val | byte5_val | byte6_val | byte7_val;
        extracted_id = result & 0x1FFFFFFF;  // Mask to 29 bits
    }

    return extracted_id;
}

9. CAN MAILBOX MODE/DLC CONFIGURATION (Address 0xCDC4)
================================================================================

/*
 * can_set_mailbox_mode_dlc(uint8_t mailbox_id, uint8_t dlc_value)
 *
 * PURPOSE:
 *   Configure mailbox mode and data length code (DLC). Sets the number
 *   of data bytes to transmit/receive (0-8) and frame type flags.
 *   Called during mailbox initialization before message transmission.
 *
 * INPUTS:
 *   R4 = mailbox_id (0-15, selects target mailbox)
 *   R5 = dlc_value (0-8, or mode bits in upper nibble)
 *
 * OUTPUTS:
 *   Side effects: modifies HCAN0_MCR_B register
 *   Bits [5:2] = DLC field (shifted version of dlc_value)
 *   Bits [1:0] = preserved, bits [7:6] = preserved
 *
 * MAILBOX REGISTERS:
 *   HCAN0_MCR_B : Mailbox control register
 *   Bit 0:      IDE (Identifier Extension) = 0 for standard, 1 for extended
 *   Bits [5:2]: DLC (Data Length Code) = shifted dlc_value
 *   Bit 6:      Control flag
 *   Bit 7:      RTR (Remote Transmission Request)
 *
 * ALGORITHM:
 *   1. Save dlc_value to stack
 *   2. Call can_get_mailbox_offset_low() to get MCR register address
 *   3. Clear FPU context before SFR operations
 *   4. Read current MCR byte
 *   5. Clear bits [3:2] using AND with 0xFB (11111011)
 *   6. Shift dlc_value left 2 bits
 *   7. OR shifted value into MCR byte
 *   8. Write modified byte back to MCR
 *   9. Clear FPU context after operations
 *
 * BIT LAYOUT EXAMPLE:
 *   Input dlc_value = 8 (8 bytes)
 *   Shift left 2: 8 << 2 = 0x20 (bits [5:2] = 10000)
 *   MCR after OR: ?????100 (bits [5:2] = 0010, DLC=8 bytes)
 *
 * NOTES:
 *   - DLC range: 0-8 bytes (standard CAN)
 *   - Bit shift [1:0] represents frame mode (standard vs extended)
 *   - All SFR accesses wrapped with fpu_load_zero() and fpu_nop_stub()
 *   - Called for both RX and TX mailbox setup
 *   - MCR preservation: only bits [5:2] modified, others preserved
 *   - get_mailbox_offset_low() returns address of MCR for this mailbox
 *
 */

void can_set_mailbox_mode_dlc(uint8_t mailbox_id, uint8_t dlc_value)
{
    volatile uint8_t *hcan0_mcr = (volatile uint8_t *)HCAN0_MCR_B;

    volatile uint8_t *mcr_addr;
    uint8_t current_mcr;
    uint8_t dlc_shifted;

    // === Save DLC Value ===
    uint8_t stack_dlc = dlc_value;

    // === Get MCR Register Address ===
    // Call can_get_mailbox_offset_low(mailbox_id)
    // Returns address of HCAN0_MCR register for this mailbox
    mcr_addr = can_get_mailbox_offset_low(mailbox_id);

    // Clear FPU context before SFR operations
    fpu_load_zero(stack_addr, 0xE0);

    // === Modify MCR Bits [5:2] ===
    // Read current MCR value
    current_mcr = *mcr_addr;

    // Clear bits [3:2] (DLC field bits 1:0)
    current_mcr &= 0xFB;  // AND with 11111011

    // Shift DLC value left 2 bits to align with bits [5:2]
    dlc_shifted = stack_dlc;
    dlc_shifted <<= 2;  // Multiply by 4

    // OR shifted DLC into MCR
    current_mcr |= dlc_shifted;

    // Write modified value back to MCR
    *mcr_addr = current_mcr;

    // Clear FPU context after operations
    fpu_nop_stub();
}

10. CAN RX PENDING FLAGS (Address 0xD0C0)
================================================================================

/*
 * uint16_t can_get_rx_pending_flags(uint8_t hcan_selector)
 *
 * PURPOSE:
 *   Read RX pending flags from HCAN0 or HCAN1, with mailbox interrupt
 *   masking applied. Indicates which mailboxes have received messages
 *   that are unmasked and pending processing.
 *
 * INPUTS:
 *   R4 = hcan_selector (0 = HCAN0, non-zero = HCAN1)
 *
 * OUTPUTS:
 *   R0 = 16-bit pending flags (each bit = one mailbox with pending RX)
 *        Bit format (mailbox 0-15): 1 = message received, 0 = no message
 *        Masked by inverse of MBIMR (mailbox interrupt mask register)
 *
 * REGISTERS (HCAN0):
 *   HCAN0_MBIMR_W : Mailbox interrupt mask register
 *                   1 = mailbox interrupt disabled, 0 = enabled
 *   HCAN0_RXPR_W  : RX pending register
 *                   1 = message received in mailbox, 0 = empty
 *
 * REGISTERS (HCAN1):
 *   HCAN1_MBIMR_W : Mailbox interrupt mask register (HCAN1)
 *   HCAN1_RXPR_W  : RX pending register (HCAN1)
 *
 * ALGORITHM:
 *   1. Check hcan_selector (R4)
 *   2. IF selector == 0:
 *        a. Read HCAN0_MBIMR_W (mailbox interrupt mask)
 *        b. NOT the value (invert bits)
 *        c. Read HCAN0_RXPR_W (RX pending)
 *        d. AND the two values (mask by inverse of MBIMR)
 *   3. ELSE (selector != 0):
 *        a. Read HCAN1_MBIMR_W
 *        b. NOT the value
 *        c. Read HCAN1_RXPR_W
 *        d. AND the two values
 *   4. Perform byte-swapping for big-endian result
 *   5. Return masked pending flags
 *
 * BIG-ENDIAN BYTE SWAP:
 *   - Input: 16-bit word [byte_H, byte_L]
 *   - Output: [byte_L, byte_H] with 8-bit shifts
 *   - extu.b r5, r3     : extract low byte
 *   - shll8 r3          : shift low byte to high position
 *   - and r5, r2        : mask high byte bits
 *   - shlr8 r2          : shift high byte to low position
 *   - or r3, r4         : combine swapped bytes
 *
 * RETURN FORMAT:
 *   Bits [7:0]    = mailbox 7-0 pending (swapped from RXPR[7:0])
 *   Bits [15:8]   = mailbox 15-8 pending (swapped from RXPR[15:8])
 *   Only unmasked mailboxes (MBIMR bit = 0) will show pending
 *
 * NOTES:
 *   - Mailbox interrupt mask inverted: NOT(MBIMR) allows unmasked mailboxes
 *   - Both HCAN0 and HCAN1 supported with selector
 *   - 16-bit register access with big-endian byte swap
 *   - Used by CAN RX interrupt handler to determine which mailbox to service
 *   - Typically called in loop to poll all pending messages
 *
 */

uint16_t can_get_rx_pending_flags(uint8_t hcan_selector)
{
    volatile uint16_t *hcan0_mbimr = (volatile uint16_t *)HCAN0_MBIMR_W;
    volatile uint16_t *hcan0_rxpr = (volatile uint16_t *)HCAN0_RXPR_W;
    volatile uint16_t *hcan1_mbimr = (volatile uint16_t *)HCAN1_MBIMR_W;
    volatile uint16_t *hcan1_rxpr = (volatile uint16_t *)HCAN1_RXPR_W;

    uint16_t mask, pending, result;
    uint16_t byte_high, byte_low;

    // === Select HCAN0 or HCAN1 ===
    if (hcan_selector == 0) {
        // HCAN0 path
        mask = *hcan0_mbimr;
        mask = ~mask;  // NOT (invert all bits)
        pending = *hcan0_rxpr;
    } else {
        // HCAN1 path
        mask = *hcan1_mbimr;
        mask = ~mask;  // NOT (invert all bits)
        pending = *hcan1_rxpr;
    }

    // === Apply Mailbox Interrupt Mask ===
    // AND with inverted mask to get only unmasked pending mailboxes
    result = pending & mask;

    // === Big-Endian Byte Swap ===
    // Result is in [byte_H | byte_L] format
    byte_low = result & 0x00FF;   // Extract low byte
    byte_high = result & 0xFF00;  // Extract high byte

    byte_low <<= 8;   // Shift low byte to high position
    byte_high >>= 8;  // Shift high byte to low position

    // === Return Swapped Result ===
    uint16_t swapped = (byte_low & 0xFF00) | (byte_high & 0x00FF);
    return swapped;
}


================================================================================
11. CAN ENABLE MAILBOX INTERRUPT (Address 0xCC6C)
================================================================================

/*
 * can_enable_mailbox_int(uint8_t hcan_selector)
 *
 * PURPOSE:
 *   Enable interrupt for a CAN mailbox. Sets mailbox interrupt enable bit
 *   in the GSR (Global Status Register) and polls until enabled.
 *   Used to arm mailbox for RX/TX event notification.
 *
 * INPUTS:
 *   R4 = hcan_selector (0 = HCAN0, non-zero = HCAN1)
 *
 * OUTPUTS:
 *   Side effects: sets bit 0 in HCAN0_GSR_B or HCAN1_GSR_B
 *   Returns after interrupt enable is confirmed
 *
 * REGISTERS:
 *   HCAN0_GSR_B : HCAN0 Global Status Register
 *   HCAN1_GSR_B : HCAN1 Global Status Register
 *   Bit 0:       Mailbox interrupt enable
 *
 * ALGORITHM:
 *   1. Check hcan_selector (R4)
 *   2. IF selector == 0:
 *        a. Use HCAN0_GSR_B address
 *   3. ELSE:
 *        a. Use HCAN1_GSR_B address
 *   4. Jump to trampoline at 0x9E14 with GSR address in R5
 *   5. At 0x9E14:
 *        a. Save GSR address to stack
 *        b. Clear FPU context
 *        c. Set bit 0 of GSR byte (OR with 0x01)
 *        d. Poll GSR bit 0 in loop until set (max 180 iterations)
 *        e. Return to caller
 *
 * JUMP TRAMPOLINE at 0x9E14:
 *   - Re-entrant service routine for both HCAN0/HCAN1 GSR access
 *   - Eliminates code duplication for nearly-identical register paths
 *   - Saves 20+ bytes of ROM space
 *   - Called with R5 pointing to target GSR register
 *
 * POLLING LOOP:
 *   - Reads GSR byte in loop
 *   - Tests bit 0 (0x08 mask for 8-bit mode)
 *   - Decrements counter from 0xB4 (180 decimal)
 *   - If bit not set after 180 tries, exits with timeout
 *
 * NOTES:
 *   - FPU context saved/restored around SFR access
 *   - GSR bit 0 controls mailbox interrupt enable globally
 *   - Individual mailbox enables controlled elsewhere
 *   - Big-endian register requires byte-aligned access
 *   - Polling provides synchronization guarantee
 *   - Used during CAN initialization and mode changes
 *
 */

void can_enable_mailbox_int(uint8_t hcan_selector)
{
    volatile uint8_t *hcan0_gsr = (volatile uint8_t *)HCAN0_GSR_B;
    volatile uint8_t *hcan1_gsr = (volatile uint8_t *)HCAN1_GSR_B;

    volatile uint8_t *target_gsr;
    uint8_t gsr_value;
    int poll_count = 180;  // 0xB4

    // === Select GSR Register ===
    if (hcan_selector == 0) {
        target_gsr = hcan0_gsr;
    } else {
        target_gsr = hcan1_gsr;
    }

    // === Save and Clear FPU Context ===
    fpu_load_zero(stack_frame, 0xE0);

    // === Set Mailbox Interrupt Enable Bit ===
    // OR bit 0 of GSR (bit 0 = mailbox interrupt enable)
    gsr_value = *target_gsr;
    gsr_value |= 0x01;
    *target_gsr = gsr_value;

    // === Clear FPU Context After Set ===
    fpu_nop_stub();

    // === Poll Mailbox Interrupt Enable ===
    // Verify that bit 0 is set (max 180 polls)
    while (poll_count > 0) {
        gsr_value = *target_gsr;
        gsr_value &= 0xFF;

        // Test bit 0 (0x08 mask in some encodings)
        if ((gsr_value & 0x08) != 0) {
            // Bit is set, interrupt enabled
            break;
        }

        poll_count--;
    }
}

// FUNCTION: knock_detection_processing (0xF770) SIZE: 0x320 | NAME VERIFIED: YES
// MAIN KNOCK DETECTION ENGINE - Per-channel knock analysis with pulse window comparison
// The rotary has 2 rotors with 3 faces each, so knock patterns differ from piston engines.
//
// INPUTS:
//   - r4 = rotor channel index (0-3, maps to rotor face)
//   - RAM base: 0xFFFFA4B4 (per-channel knock state structures, 0x58 bytes each)
//   - Knock enabled flag: 0xFFFFA571 (1=enabled, 0=disabled)
//
// ROM CALIBRATION:
//   - Period multiplier: 0x1E0000
//   - Detection threshold constant: 0x1680000 (baseline knock threshold)
//   - Timing lookup: 0x4ECE0, VVT mask table: 0x4ECC0
//   - Per-channel config: 0x4ECE4 (state machine config)
//
// GLOBAL COUNTER: 0xFFFFA564 (u16) - saturated knock event counter
//
// ALGORITHM:
//   Phase 1: Normalize rotor channel index, compute period offset = ch * 0x1E0000
//   Phase 2: Check 0xFFFFA571 enable flag. If disabled, clear all state.
//   Phase 3: Init per-channel structures (2 × 0x58 bytes), set threshold = 0x1680000
//   Phase 4: First pass - measure knock via pulse_window_compute(), find max level,
//            compare vs threshold. If exceeded: increment counter, compute margin.
//   Phase 5: Monitor IACV duty and VVT solenoid status for timing adjustments.
//   Phase 6: Second pass - state machine (0=monitoring, 1=retard active, 2=recovery)
//
// TUNER IMPACT:
//   - Threshold 0x1680000: INCREASING = less sensitive, DECREASING = more sensitive
//   - Margin divisor 0xB40000: affects retard magnitude per knock event
void knock_detection_processing(uint8_t channel_index) {
  uint8_t normalized_ch = channel_index & 0xFF;
  uint8_t last_channel = read_byte(0xFFFFA570);
  int8_t delta = normalized_ch - last_channel;
  write_byte(0xFFFFA570, channel_index);
  if (delta < 0) delta += 0x18;
  if (normalized_ch >= 12) normalized_ch -= 12;
  
  uint32_t period_offset = normalized_ch * 0x1E0000;
  uint8_t* knock_base = (uint8_t*)0xFFFFA4B4;
  uint16_t* knock_counter = (uint16_t*)0xFFFFA564;
  uint32_t threshold = 0x1680000;
  
  if (read_byte(0xFFFFA571) != 1) {
    if (delta != 1) injection_pulse_decoder_main();
    goto read_sensors;
  }
  
  // Clear all knock state
  write_byte(0xFFFFA571, 0);
  write_byte(0xFFFFA56D, 0);
  write_word(0xFFFFA56E, 0);
  write_word(0xFFFFA564, 0);
  write_dword(0xFFFFA568, 0);
  
  // Init per-channel structures
  for (uint8_t* ch = knock_base; ch < knock_base + 0xB0; ch += 0x58) {
    memset(ch, 0, 8);
    for (uint8_t* sub = ch + 0x10; sub < ch + 0x58; sub += 0x18) {
      *(uint32_t*)(sub+4) = 0;
      *(uint32_t*)(sub+8) = 0;
      *(uint32_t*)(sub+0xC) = threshold;
      *(uint32_t*)(sub+0x10) = threshold;
    }
  }
  
  // First pass: measure knock levels
  for (uint8_t* ch = knock_base; ch < knock_base + 0xB0; ch += 0x58) {
    uint32_t pulse = pulse_window_compute(period_offset, *(uint32_t*)(ch+8));
    fuel_injector_multiplexed_control(ch);
    uint32_t max_level = 0;
    for (uint8_t* sub = ch + 0x10; sub < ch + 0x58; sub += 0x18)
      if (*(uint32_t*)(sub+0x10) > max_level) max_level = *(uint32_t*)(sub+0x10);
    
    if (pulse >= max_level) {  // KNOCK DETECTED
      *knock_counter = saturated_add_u16(*knock_counter, 1);
      *(uint16_t*)(ch+4) = (pulse - max_level) / 0xB40000 + 1;
      ch[3] = 1; ch[2] = 0;
    } else {
      *(uint16_t*)(ch+4) = 0; ch[3] = 0; ch[2] = 2;
    }
  }
  
read_sensors:
  uint8_t iacv = read_iacv_duty();
  if (iacv != read_byte(0xFFFFA56D)) {
    injection_timing_calculator(period_offset, ROM_BYTE(0x4ECE0 + iacv));
    write_byte(0xFFFFA56D, iacv);
  }
  uint16_t vvt = read_vvt_solenoid_status();
  if (vvt != read_word(0xFFFFA56E)) {
    injection_timing_filter_mask(period_offset, vvt);
    write_word(0xFFFFA56E, vvt);
  }
  
  // Second pass: state machine response
  for (uint8_t* ch = knock_base; ch < knock_base + 0xB0; ch += 0x58) {
    uint32_t pulse = pulse_window_compute(period_offset, *(uint32_t*)(ch+8));
    uint32_t* config = (uint32_t*)(0x4ECE4 + ch[0] * 16);
    switch (ch[2]) {
      case 0: // MONITORING - check for retard need
        if (pulse > config[1] && pulse <= config[0] && ch[3] == 0) {
          *knock_counter = saturated_add_u16(*knock_counter, 1);
          ch[3] = 1;
        }
        break;
      case 1: // RETARD ACTIVE - call response handler
        if (ch[1] == 0 && ch[6] == 1) {
          ((void(*)(uint8_t*,uint32_t))config[2])(ch, pulse);
        }
        break;
      case 2: break; // RECOVERY - no action
    }
  }
}

// FUNCTION: cold_start_rpm_limiter (0xF11A) SIZE: 0x78 | NAME VERIFIED: YES
// Applies reduced RPM ceiling during engine warm-up for apex seal protection.
// ROM: Cold threshold 0x6D53C, hysteresis 0x6D540, duty value 0x6D534
// RAM: 0xFFFFA4A6 (active flag), 0xFFFFA4AE (duty counter), 0xFFFFA4AA (fuel cut)
void cold_start_rpm_limiter(void) {
  float rpm = read_float(0xFFFFAA10);
  float cold_thresh = read_float(0x6D53C);
  float hysteresis = read_float(0x6D540);
  float warm_thresh = cold_thresh - hysteresis;
  
  if (rpm > warm_thresh) write_byte(0xFFFFA4A6, 1);
  else if (rpm <= cold_thresh) write_byte(0xFFFFA4A6, 0);
  
  if (read_byte(0xFFFFB5A4) == 0 && read_byte(0xFFFFBB54) != 0 &&
      read_byte(0xFFFFA4AD) == 0 && read_byte(0xFFFFBB54) != 7) {
    write_byte(0xFFFFA4AE, ROM_BYTE(0x6D534));
  }
  
  if (read_byte(0xFFFFA4AE) > 0) {
    write_byte(0xFFFFA4AA, 1);  // FUEL CUT
    write_byte(0xFFFFA4AE, read_byte(0xFFFFA4AE) - 1);
  } else {
    write_byte(0xFFFFA4AA, 0);  // Normal
  }
  write_byte(0xFFFFA4AD, read_byte(0xFFFFBB54));
}


// FUNCTION: rpm_limiter_main (0xF192) SIZE: 0x1C4 | NAME VERIFIED: YES
// PRIMARY REV LIMITER - Multi-tiered: soft limit (fuel reduction), hard limit (fuel cut)
// ROM: Rev_Limit 0x6D54C, hard offset 0x6D550, Cold_Rev_Limit 0x6D53C
// RAM: 0xFFFFA4A4 (active), 0xFFFFA4A5 (mode), 0xFFFFA4AB (fuel cut), 0xFFFFA4AC (hard)
// Duration counter: 0xFFFFA4A8 (u16, saturated), thresholds: 0x6D536, 0x6D538
void rpm_limiter_main(void) {
  float rpm = read_float(0xFFFFB5B8);
  float soft = read_float(0x6D54C);        // Rev_Limit
  float hard_off = read_float(0x6D550);
  float cold = read_float(0x6D53C);
  float cold_hyst = read_float(0x6D544);
  
  // Soft limit with hysteresis
  if (rpm > cold) write_byte(0xFFFFA4AB, 1);
  else if (rpm <= cold - cold_hyst) write_byte(0xFFFFA4AB, 0);
  
  // Hard limit
  if (rpm > soft) write_byte(0xFFFFA4AC, 1);
  else if (rpm <= soft - hard_off) write_byte(0xFFFFA4AC, 0);
  
  uint8_t any_active = read_byte(0xFFFFA4A6) || read_byte(0xFFFFA4AA) ||
                        read_byte(0xFFFFA4AC);
  if (!any_active && !read_byte(0xFFFFA4AB)) {
    write_byte(0xFFFFA4A5, 0);
    write_byte(0xFFFFA4A4, 0);
    write_word(0xFFFFA4A8, 0);
    return;
  }
  
  write_byte(0xFFFFA4A4, 1);
  uint16_t dur = saturated_add_u16(read_word(0xFFFFA4A8), 1);
  write_word(0xFFFFA4A8, dur);
  
  // Duration safety checks
  if (read_byte(0xFFFFB5AA) == 0) {
    if (dur > read_word(0x6D536)) write_byte(0xFFFFA4A5, 0);  // Force clear
    if (dur > read_word(0x6D538)) write_byte(0xFFFFA4A5, 0);  // Hard timeout
  }
}


// FUNCTION: rpm_limiter_fuel_cutoff (0xC59E) SIZE: 0xB8 | NAME VERIFIED: YES
// Converts RPM overspeed error to fuel injector duty cycle for smooth cutoff.
// ROM: Enable threshold 0x7A1D8, disable 0x7A1DC, control ref 0x7A1DE
// Output: 0xFFFFA386 (fuel cut duty byte)
void rpm_limiter_fuel_cutoff(void) {
  float rpm = read_float(0xFFFF9F80);
  write_word(0xFFFFA37A, read_word(0xFFFF9EF8));
  
  float enable = read_float(0x7A1D8);
  float disable = read_float(0x7A1DC);
  
  if (rpm <= enable && rpm >= disable) {
    float err = fpu_abs_float((float)read_word(0xFFFFA37A));
    write_float(0xFFFFA374, err);
    write_word(0xFFFFA378, (uint16_t)err);
  }
  
  if (rpm > read_float(0x7A198)) { /* above range */ }
  else { write_byte(0xFFFFA386, 1); return; }  // FUEL CUT
  
  if (rpm >= read_float(0x7A194)) { /* below range */ }
  else { write_byte(0xFFFFA386, 0); }  // Normal
}

// FUNCTION: lambda_control_closed_loop (0xC508) SIZE: 0x482 | NAME VERIFIED: YES
// PRIMARY O2 FEEDBACK CONTROLLER - PI fuel trim using dual narrowband O2 sensors
// RX-8 has 2 narrowband O2 sensors (one per rotor) + 2 post-cat sensors.
//
// ROM PID PARAMETERS:
//   P gain: 0x6B8D4 | I gain #1: 0x6B90C | I gain #2: 0x6B900
//   P clamp: 0xFFFFC0CC | I step: 0xFFFFC0D0 | Total clamp: 0xFFFFC0D4
//
// RAM STATE:
//   0xFFFFA370: P error | 0xFFFFA330: gated P | 0xFFFFA32C: filtered delta
//   0xFFFFA348: prev setpoint | 0xFFFFA350: integral accum array
//   0xFFFFA358/35C/360: integrator outputs | 0xFFFFA364: final correction
//   0xFFFFA368: per-rotor output | 0xFFFFA324: CL status (0=OL, 1=CL)
//   0xFFFFA381: threshold multiplier (×20 scale per rotor)
//   0xFFFFA38C: disable flag (1=open-loop)
void lambda_control_closed_loop(uint8_t cyl_event) {
  if (read_byte(0xFFFFA38C)) return;  // Open-loop mode
  
  uint8_t cyl = cyl_event >= 12 ? cyl_event - 12 : cyl_event;

  // Match rotor to O2 sensor pair
  uint8_t match = find_o2_sensor_pair(cyl, 0xDBA6, 0xDBA8);
  if (match == 1) {
    atu_schedule_event_from_adc();
    if (read_byte(0xFFFFA384) == 0xFF) return;
  } else if (match == 2) {
    lambda_error_computation();
    write_byte(0xFFFFA385, 0);
    return;
  } else return;
  
  // === MAIN PID (0xC656) ===
  
  // 1. PROPORTIONAL
  float err = (float)(read_word(0xFFFFA378) - read_word(0xFFFFA326));
  float p_raw = err * read_float(0xC8B4);
  if (p_raw < 0) p_raw = 0;
  float p_term = fpu_multiply_accumulate(0x6B8D4, p_raw);
  write_float(0xFFFFA370, p_term);
  
  // 2. THRESHOLD GATE (deadband)
  uint8_t idx = read_byte(0xFFFFA384);
  float threshold = (float)(read_byte(0xFFFFA381 + idx) << 5);
  float gated = max(0, p_term - threshold);
  write_float(0xFFFFA330, gated);
  
  // 3. DERIVATIVE ANALOG
  float delta = gated - read_float(0xFFFFA348);
  write_float(0xFFFFA32C, fpu_abs_float(delta));

  // 4. PER-ROTOR AFR CORRECTION
  float* afr = (float*)0xFFFFA334;
  float corr = read_float(0xFFFFA344);
  float* trim = (float*)0xFFFFA33C;
  if (gated > afr[idx]) trim[idx] += corr;
  else if (gated < afr[idx]) trim[idx] -= corr;
  if (trim[idx] < 0) trim[idx] = 0;
  afr[idx] = trim[idx];
  
  // 5. PRIMARY INTEGRATOR
  float rpm = read_float(0xFFFF9F80);
  float i_primary = fpu_multiply_accumulate(0x6B90C, rpm);
  write_float(0xFFFFA358, i_primary);
  
  // 6. SECONDARY INTEGRATOR
  float i_secondary = fpu_multiply_accumulate(0x6B900, rpm);
  float* accum = (float*)0xFFFFA350;
  float step = corr * read_float(0xFFFFC0D0);
  accum[idx] += (gated > afr[idx]) ? -step : step;
  accum[idx] = clamp(accum[idx], i_secondary, i_primary);
  
  // 7. FINAL OUTPUT
  float final = fmac(read_float(0xFFFFA360), read_float(0xFFFFA328),
                      read_float(0xFFFFA364));
  final = clamp(final, read_float(0x7A1D0), read_float(0x7A1D4));
  write_float(0xFFFFA364, final);
  
  write_byte(0xFFFFA324, (final > gated) ? 0 : 1);  // CL status
}


// FUNCTION: ignition_advance_limiter (0xE38C) SIZE: 0xB0 | NAME VERIFIED: YES
// Final safety valve: prevents advance overflow, enforces hard limits.
// CHECK 1: Safe mode (0xFFFFA42C) → force 2° advance
// CHECK 2: Counter overflow (0xFFFFA416 vs Table_2D_331+0x34)
// CHECK 3: Accel enrichment (0xFFFFB5B8 vs Table_2D_331+0x3C) → increment counter
void ignition_advance_limiter(void) {
  float load = read_float(0xFFFFAA10);
  
  if (load > read_float(TABLE_2D_331 + 0x38) && read_byte(0xFFFFA42C) == 1) {
    write_byte(0xFFFFA418, 2);  // Force 2° advance (minimum safe)
    crank_set_state_byte(1);
    goto check_accel;
  }
  
  if (read_word(0xFFFFA416) >= read_word(TABLE_2D_331 + 0x34) ||
      read_byte(0xFFFFA41C) == 1) {
    write_byte(0xFFFFA418, 1);
    crank_set_state_byte(0);
  } else {
    write_byte(0xFFFFA418, 0);
  }
  
check_accel:
  if (read_float(0xFFFFB5B8) > read_float(TABLE_2D_331 + 0x3C))
    write_word(0xFFFFA416, saturated_add_u16(read_word(0xFFFFA416), 1));
  else
    write_word(0xFFFFA416, 0);
}

NOTES FOR REVERSE ENGINEERING
================================================================================

1. The SH-2A architecture uses 16-bit instructions (mostly). All addresses
   are byte-addressed but code runs in 2-byte alignment.

2. Registers R14, R13, R12, R11 are call-saved (preserved by callee).
   Registers R0-R7 are call-clobbered.

3. The 0x5A unlock key for WDT registers is critical. Writing wrong key
   causes register writes to be ignored.

4. The warm-reset magic (0x5AA5A55A) serves as a flag to skip RAM
   initialization on warm restarts, speeding up reset handling.

5. Task scheduler uses priority levels in SR[7:4]. Higher numeric value
   = higher interrupt priority (0xF0 is highest, 0x00 is lowest).

6. The task control block at 0xFFFF72B0 contains:
   - Offset 0x04: Queue write pointer
   - Offset 0x06: Queue read pointer
   - Offset 0x08: Work flag (0x100 = task flag set, 0x1000 = work pending)
   - Offset 0x10: Task priority level
   - Offset 0x18: Pointer to task state structure

7. The handler at 0x3DB0 (within task_conditional_dispatch function)
   performs full context save/restore for task switching.

8. Fuel injection and ignition timing are handled by high-priority
   scheduler at 0x3A6EA, called from engine interrupt handlers.
